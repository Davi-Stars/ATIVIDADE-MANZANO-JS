// Exerciço 3.10-C: O dia da eleição! Vamo contar esses votos
let votosA = parseInt(prompt("Votos para A:"));
let votosB = parseInt(prompt("Votos para B:"));
let votosC = parseInt(prompt("Votos para C:"));
let nulos = parseInt(prompt("Votos nulos:"));
let brancos = parseInt(prompt("Votos em branco:"));
let total = votosA + votosB + votosC + nulos + brancos;
let percVotos = (votosA + votosB + votosC) / total * 100;
let percA = votosA / total * 100;
let percB = votosB / total * 100;
let percC = votosC / total * 100;
let percNulos = nulos / total * 100;
let percBrancos = brancos / total * 100;
console.log("Total de eleitores: ", total);
console.log("Percentual de votos válidos: ", percVotos.toFixed(2) + "%");
console.log("Percentual de votos A: ", percA.toFixed(2) + "%");
console.log("Percentual de votos B: ", percB.toFixed(2) + "%");
console.log("Percentual de votos C: ", percC.toFixed(2) + "%");
console.log("Percentual de votos nulos: ", percNulos.toFixed(2) + "%");
console.log("Percentual de votos brancos: ", percBrancos.toFixed(2) + "%");  // Olha aí as parciais...
